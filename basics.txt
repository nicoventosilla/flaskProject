from flask import Flask, request, render_template, url_for, abort, session
from werkzeug.utils import redirect

app = Flask(__name__)

app.secret_key = 'clave_secreta'


@app.route('/')
def index():
    if 'username' in session:
        return f'Hello {session["username"]}!'
    return f'Usuario no identificado'


@app.route('/login', methods=['GET', 'POST'])
def login():
    if request.method == 'POST':
        session['username'] = request.form['username']
        return redirect(url_for('index'))
    return render_template('login.html')


@app.route('/logout')
def logout():
    session.pop('username', None)
    return redirect(url_for('index'))


@app.route('/saludo/<nombre>')
def saludo(nombre):
    return f'Hola {nombre.title()}!'


@app.route('/edad/<int:edad>')
def edad(edad):
    return f'Tienes {edad} a√±os'


@app.route('/mostrar/<nombre>', methods=['GET', 'POST'])
def mostrar(nombre):
    return render_template('mostrar.html', nombre=nombre)


@app.route('/redireccionar')
def redireccionar():
    return redirect(url_for('saludo', nombre='Pepito'))


@app.route('/salir')
def salir():
    return abort(404)


@app.errorhandler(404)
def pagina_no_encontrada(error):
    return render_template('404.html', error=error), 404


# REST API
@app.route('/api/suma', methods=['POST'])
def suma():
    try:
        datos = request.get_json()
        suma = datos['a'] + datos['b']
        return {'resultado': suma}
    except KeyError:
        return {'error': 'Faltan datos'}, 400
    except TypeError:
        return {'error': 'Datos incorrectos'}, 400


if __name__ == '__main__':
    app.run()
